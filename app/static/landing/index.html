3 Lines. Full AI Pipeline.

No complex setup. No boilerplate. Just snap blocks together.

from cerebrum import CerebrumClient, chain

# Initialize client
client = CerebrumClient("http://localhost:8000", "your-api-key")

# Chain blocks together
result = await chain(client) \
    .then("pdf", {"extract": "text"}) \
    .then("translate", {"target": "en"}) \
    .then("chat", {"prompt": "Summarize this contract:"}) \
    .run("contract.pdf")

print(result.final_output)
